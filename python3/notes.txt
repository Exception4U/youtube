# copy/slicing lists
# example of odd/even results:
x = list(range(1,31))

# print elements 3 through 10
print (x[2:10])

# print out the last element
print (x[len(x)-1])
# better way
print (x[-1])

# print everything except the last element
print (x[:-1])

print ('odd values are: ' + str(x[::2]))
print ('even values are: ' + str(x[1::2]))
